# Dados de microbiota pr e ps exposio ao antibitico ATB1329A e placebo.
# P1 = Primeira anlise populacional de microbiota
# P2 = Segunda anlise populacional de microbiota (3 meses aps a primeira analise)

# Cenrio 1 - Comparao Temporal - Geral

# Reprodutibilidade
set.seed(129) 

# --- Tamanho amostral ---
n <- 92 

# --- Gerar o P1 --- # Pr-tratamento
media1 <- 1000       
sd1 <- 150           
P1 <- rnorm(n, mean = media1, sd = sd1)

# --- Gerar o P2 --- # Ps-tratamento
b0 <- 100 
b1 <- 0.8        
P2 <- b0 + b1*P1

# --- Gerar o Data frame ---
dados <- data.frame(P1, P2)

# --- Gerar o lineplot ---

library(sciplot)

dados_long <- data.frame(
  Grupo = rep(c("P1 (Pr-tratamento)", "P2 (Ps-tratamento)"), each = n),
  Valor = c(dados$P1, dados$P2)
)
par(mfrow=c(1,1))
lineplot.CI(x.factor = dados_long$Grupo,
            response = dados_long$Valor,
            las = 1,
            type = "p",
            xlab = "Momento",
            ylab = "Valores da microbiota",
            main = "Comparao de Mdias: P1 x P2",
            ci.fun = function(x) c(mean(x)-sd(x), mean(x)+sd(x)),
            pch = 19,
            col = c("blue", "red")) 

# --- Gerar o Boxplot ---
